    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Gazastreifen auf der Schweizer Karte</title>

    

    <link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css" />

    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>

    

    <script src="https://cdn.jsdelivr.net/npm/leaflet-path-drag@1.1.0/dist/L.Path.Drag.min.js"></script>


    <style>

        /* Stellt sicher, dass die Karte den ganzen Bildschirm füllt */

        body { margin: 0; padding: 0; }

        #map { position: absolute; top: 0; bottom: 0; width: 100%; }

    </style>

</head>

<body>


    <div id="map"></div>


    <script>

        // 1. Die Karte initialisieren und auf die Schweiz zentrieren

        // Die Zahlen in setView sind [Breitengrad, Längengrad] und der Zoom-Level

        var map = L.map('map').setView([46.8182, 8.2275], 8);


        // 2. Die Hintergrundkarte hinzufügen (hier eine detaillierte Schweizer Karte)

        L.tileLayer('https://tile.openstreetmap.ch/switzerland/{z}/{x}/{y}.png', {

            maxZoom: 18,

            attribution: '&copy; <a href="http://www.openstreetmap.org/copyright">OpenStreetMap</a>'

        }).addTo(map);


        // 3. Die Geodaten für den Gazastreifen (Umriss und Städte)

        // Die Daten sind im GeoJSON-Format direkt hier im Code hinterlegt.

        var gazaGeoJSON = {

          "type": "FeatureCollection",

          "features": [

            {

              "type": "Feature",

              "properties": { "name": "Gazastreifen" },

              "geometry": {

                "type": "Polygon",

                "coordinates": [[

                  [34.33232, 31.22502], [34.22669, 31.30959], [34.24803, 31.36592], 

                  [34.2205, 31.42225], [34.28292, 31.49257], [34.34533, 31.54225], 

                  [34.41727, 31.56592], [34.48292, 31.59624], [34.54227, 31.61657], 

                  [34.56361, 31.59959], [34.46903, 31.50592], [34.40661, 31.43225], 

                  [34.35824, 31.33259], [34.33232, 31.22502]

                ]]

              }

            },

            { "type": "Feature", "properties": { "name": "Gaza-Stadt" }, "geometry": { "type": "Point", "coordinates": [34.4667, 31.5] } },

            { "type": "Feature", "properties": { "name": "Chan Yunis" }, "geometry": { "type": "Point", "coordinates": [34.3, 31.35] } },

            { "type": "Feature", "properties": { "name": "Rafah" }, "geometry": { "type": "Point", "coordinates": [34.25, 31.2833] } }

          ]

        };


        // 4. Den Gazastreifen auf der Karte platzieren und gestalten

        // Wir starten ihn über der Region Genf als Beispiel

        var gazaLayer = L.geoJSON(gazaGeoJSON, {

            style: function(feature) {

                if (feature.geometry.type === 'Polygon') {

                    return {

                        color: "#e60000", // Rote Umrisslinie

                        weight: 2,

                        fillColor: "#e60000",

                        fillOpacity: 0.4 // Halbtransparent

                    };

                }

            },

            // Fügt den Städtenamen als kleinen Tooltip hinzu

            onEachFeature: function(feature, layer) {

                if (feature.properties && feature.properties.name) {

                    layer.bindTooltip(feature.properties.name, {

                        permanent: true, // Name immer anzeigen

                        direction: 'center',

                        className: 'gaza-label' // CSS-Klasse für optionales Styling

                    });

                }

            },

            // Stellt sicher, dass die Städte als einfache Kreise dargestellt werden

            pointToLayer: function(feature, latlng) {

                return L.circleMarker(latlng, {

                    radius: 4,

                    fillColor: "#ffffff",

                    color: "#000",

                    weight: 1,

                    opacity: 1,

                    fillOpacity: 0.8

                });

            }

        }).addTo(map);


        // 5. Die Funktion zum Verschieben für die ganze Ebene aktivieren

        gazaLayer.dragging.enable();


    </script>

</body>
